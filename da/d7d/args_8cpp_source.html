164bool str_to_bool(const char* str) {
165 char* lower = (char*)malloc(strlen(str) + 1);
166 str_to_lower(str, lower);
167 if(strcmp(lower, "on") == 0 ||
168 strcmp(lower, "true") == 0) {
169 free(lower);
170 return true;
171 }
172
173 if(strcmp(lower, "off") == 0 ||
174 strcmp(lower, "false") == 0) {
175 free(lower);
176 return false;
177 }
178
179 free(lower);

227bool find_arg(const char* name, Args::Arg* out) {
228 if(name == NULL || out == NULL) {
229 return false;
230 }
231
232 // Strip the expected argument prefix and then compare against
233 // canonical long/short argument names.
234 const char* arg = nullptr;
235 bool kind = false;
236
237#if defined(_WIN32) || defined(_WIN64)
238 if(starts_with(WS_CLI_ARG_LONG_BEGIN, name) || starts_with(WS_CLI_ARG_SHORT_BEGIN, name)) {
239 arg = name + 1;
240 } else {
241 return false;
242 }
243#else
244 if(starts_with(WS_CLI_ARG_LONG_BEGIN, name)) {
245 kind = true;
246 arg = name + 2;
247 } else if(starts_with(WS_CLI_ARG_SHORT_BEGIN, name)) {
248 arg = name + 1;
249 } else {
250 return false;
251 }
252#endif
253
254 for(int i = 0; i < Args::args.size(); i++) {
255 bool is_arg = false;
256 #if defined(_WIN32) || defined(_WIN64)
257 // Completely discard "kind"
258 if(strcmp(Args::args[i].arg_long, arg) == 0 ||
259 strcmp(Args::args[i].arg_short, arg) == 0) {
260 is_arg = true;
261 }
262 #else
263 if(kind &&
264 strcmp(Args::args[i].arg_long, arg) == 0) {
265 is_arg = true;
266 } else if(!kind && strcmp(Args::args[i].arg_short, arg) == 0) {
267 is_arg = true;
268 }
269 #endif
270 if(is_arg) {
271 memcpy(out, &Args::args[i], sizeof(Args::Arg));
272 return true;
273 }
274 }
275
276 return false;
277}
